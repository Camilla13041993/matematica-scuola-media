\usepackage{environ}

\newcounter{esercizio}

% Ambiente esercizio con lo stile pulito (senza tcolorbox scuro)
\newenvironment{esercizio}[1]{%
  \needspace{9.5cm}%
  \refstepcounter{esercizio}%
  \vspace{0.25cm}%
  {\large\bfseries\color{mainblue}Esercizio \theesercizio\ -- #1}\par
  \vspace{0.1cm}%
}{\vspace{0.3cm}}

% Ambiente soluzione condizionale (si compila solo se \ifsoluzioni è vero)
\NewEnviron{soluzione}{%
  \ifsoluzioni
    \vspace{0.15cm}
    \begin{tcolorbox}[
      colback=green!5,
      colframe=green!40!black,
      title={Soluzione Esercizio \theesercizio},
      fonttitle=\bfseries,
      arc=2mm,
      boxrule=0.5pt,
      left=3mm,
      right=3mm,
      top=2mm,
      bottom=2mm
    ]
      \BODY
    \end{tcolorbox}
    \vspace{0.15cm}
  \fi
}

% Altri ambienti preesistenti per compatibilità
\newenvironment{reminder}{%
  \begin{tcolorbox}[
    colback=gray!8,
    colframe=black!30,
    title={Ricorda},
    fonttitle=\bfseries,
    arc=2mm,
    boxrule=0.5pt
  ]
}{%
  \end{tcolorbox}
}
